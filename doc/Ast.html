type binop =
| Add
| Sub
| Mult
| Div
| Mod
| Equal
| Neq
| Less
| Leq
| Greater
| Geq
| And
| Or
| Comma 
type uop =
| Neg
| Not
| Incr
| Decr 
type identifier = string 
type position = Stdlib.Lexing.position * Stdlib.Lexing.position 
val dummy_pos : Stdlib.Lexing.position * Stdlib.Lexing.position
type 'a annotated_node = {
   loc : position;
   node : 'a;
   id : int;
} 
type typ =
| TypI (*

Type int

*)
| TypF (*

Type float

*)
| TypB (*

Type bool

*)
| TypC (*

Type char

*)
| TypA of typ * int option (*

Array type

*)
| TypP of typ (*

Pointer type

*)
| TypV (*

Type void

*)
| TypFun of typ * string * typ list (*

Type function

*) 
type expr = expr_node annotated_node 
type expr_node =
| Access of access (*

x or *p or ae

*)
| Assign of access * expr (*

x=e or *p=e or ae=e

*)
| Addr of access (*

&x or &*p or &ae

*)
| ILiteral of int (*

Integer literal

*)
| FLiteral of float (*

Float literal

*)
| CLiteral of char (*

Char literal

*)
| BLiteral of bool (*

Bool literal

*)
| NLiteral of unit (*

NULL literal

*)
| UnaryOp of uop * expr (*

Unary primitive operator

*)
| BinaryOp of binop * expr * expr (*

Binary primitive operator

*)
| Call of identifier * expr list (*

Function call f(...)

*) 
type access = access_node annotated_node 
type access_node =
| AccVar of identifier (*

Variable access x

*)
| AccDeref of expr (*

Pointer dereferencing *p

*)
| AccIndex of access * expr (*

Array indexing ae

*) 
type stmt = stmt_node annotated_node 
type stmt_node =
| If of expr * stmt * stmt (*

Conditional

*)
| While of expr * stmt (*

While loop

*)
| Expr of expr (*

Expression statement e;

*)
| Return of expr option (*

Return statement

*)
| Block of stmtordec list (*

Block: grouping and scope

*) 
type stmtordec = stmtordec_node annotated_node 
type stmtordec_node =
| Dec of typ * identifier (*

Local variable declaration

*)
| Stmt of stmt (*

A statement

*) 
type fun_decl = {
   typ : typ;
   fname : string;
   formals : (typ * identifier) list;
   body : stmt;
} 
type topdecl = topdecl_node annotated_node 
type topdecl_node =
| Fundecl of fun_decl
| Vardec of typ * identifier 
type program =
| Prog of topdecl list
